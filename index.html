How to Use

<!-- On your HTML -->

<div class="row">
    <div class="col-1-3">
        <p>First column</p>
    </div>

    <div class="col-1-3">
        <p>Second column</p>
    </div>

    <div class="col-1-3">
        <p>Third column</p>
    </div>
</div> <!-- .row -->
// On your main style.less file
// ## Grid Setup
@import 'hyx';
@gutter: 4%;

  // namespace
  .col-1-3  { 
    .cols(1/3); // one of three
  }

Compiled to: 

// compiled style.css 
.col-1-3 {
  float: left;
  width: 30.666666666666668%;
  margin-right: 4%;
}
// when fractions (e.g. `1/4`) used
// this pseudo selector will be used
.col-1-3:last-child {
  margin-right: 0;
}
